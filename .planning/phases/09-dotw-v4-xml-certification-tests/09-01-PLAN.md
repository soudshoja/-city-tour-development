        $itinerary = $dotwService->saveBooking($confirmParams, $resayilMessageId, $resayilQuoteId, $companyId);

    }
} else {
    // Standard confirmbooking flow
    try {
        $confirmation = $dotwService->confirmBooking($confirmParams, $resayilMessageId, $resayilQuoteId, $companyId);
    } catch (\Exception $e) {

}
