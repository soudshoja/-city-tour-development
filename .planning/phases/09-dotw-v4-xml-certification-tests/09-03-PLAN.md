Add the following 5 methods, each following the exact same pattern as `getCountryList()`:

All 7 resolvers are nearly identical — they differ only in which DotwService method they call and which response data type they return.

Use `DotwGetCities.php` as the structural template (it is the simplest existing resolver). All follow:

```php
<?php
declare(strict_types=1);

namespace App\GraphQL\Queries;

use App\Services\DotwService;

/**
 * Resolver for {queryName} query.
 *
 * Calls DOTW {command} command and returns {data description}.
 */
class DotwGet{Name}
{
    public function __invoke(mixed $root, array $args): array
    {
        $companyId = auth()->user()?->company?->id;

        try {
            $dotwService = new DotwService($companyId);
            $items = $dotwService->{serviceMethod}();
        } catch (\Exception $e) {
            return [
                'success' => false,
                'error' => [
                    'error_code' => 'API_ERROR',
                    'error_message' => 'Failed to fetch {name}: ' . $e->getMessage(),
                    'error_details' => $e->getMessage(),
                    'action' => 'RETRY',
                ],
                'meta' => $this->buildMeta($companyId ?? 0),
                'cached' => false,
                'data' => null,
            ];
        }

        return [
            'success' => true,
            'error' => null,
            'meta' => $this->buildMeta($companyId ?? 0),
            'cached' => false,
            'data' => [
                '{dataKey}' => $items,   // countries / classifications / locations / amenities / preferences / chains
                'total_count' => count($items),
            ],
        ];
    }

    private function buildMeta(int $companyId): array
    {
        return [
            'trace_id' => app('dotw.trace_id'),
            'timestamp' => now()->toIso8601String(),
            'company_id' => $companyId,
            'request_id' => app('dotw.trace_id'),
        ];
    }
}
```
